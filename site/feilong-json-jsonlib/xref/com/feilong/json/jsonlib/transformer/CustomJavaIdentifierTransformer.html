18  import static com.feilong.core.lang.ObjectUtil.defaultIfNullOrEmpty;
19  
20  import java.util.Map;
21  
22  import org.apache.commons.lang3.Validate;

24  import net.sf.json.util.JavaIdentifierTransformer;

26  /**
27   * 当json转成bean的时候,json字符串里面的属性名字可能有部分不符合我们的java属性命名规范,此时可以基于部分属性做转换器.
28   * 
29   * <p>
30   * 比如 country_id,但是我们的java bean里面的属性名字是标准的 驼峰命名法则,比如 countryId
31   * </p>
32   * 
33   * <h3>示例:</h3>
34   * 
35   * <blockquote>
36   * 
37   * <p>
38   * <b>场景:</b> 从相关接口得到的json数据格式如下(注意:部分属性 名字中间是 下划线):
39   * </p>
40   * 
41   * <pre class="code">
42   {
43              "country": "中国",
44              "country_id": "CN",
45              "area": "华北",
46              "area_id": "100000",
47              "region": "北京市",
48              "region_id": "110000",
49              "city": "北京市",
50              "city_id": "110100",
51              "county": "",
52              "county_id": "-1",
53              "isp": "科技网",
54              "isp_id": "1000114",
55              "ip": "210.75.225.254"
56          }
57   * </pre>
58   * 
59   * <p>
60   * 但是我们的类是标准的java bean,属性符合驼峰命名规则,比如:
61   * </p>
62   * 
63   * <pre class="code">
64   * public class IpInfoEntity{
65   * 
66   *     <span style="color:green">//ip比如 210.75.225.254.</span>
67   *     private String ip;
68   * 
69   *     <span style="color:green">//** "country": "中国".</span>
70   *     private String country;
71   * 
72   *     <span style="color:green">//** "area": "华北".</span>
73   *     private String area;
74   * 
75   *     <span style="color:green">//** "region": "北京市".</span>
76   *     private String region;
77   * 
78   *     <span style="color:green">//** "city": "北京市".</span>
79   *     private String city;
80   * 
81   *     <span style="color:green">//** "isp": "电信".</span>
82   *     private String isp;
83   * 
84   *     <span style="color:green">//** "country_id": "86".</span>
85   *     private String countryId;
86   * 
87   *     <span style="color:green">//** "region_id": "110000".</span>
88   *     private String regionId;
89   * 
90   *     <span style="color:green">//** "area_id": "100000".</span>
91   *     private String areaId;
92   * 
93   *     <span style="color:green">//** "city_id": "110000".</span>
94   *     private String cityId;
95   * 
96   *     <span style="color:green">//** "isp_id": "100017".</span>
97   *     private String ispId;
98   * 
99   *     <span style="color:green">// setter /getter 方法省略</span>
100  * 
101  * }
102  * 
103  * </pre>
104  * 
105  * <p>
106  * 此时可以使用该类,示例如下:
107  * </p>
108  * 
109  * <pre class="code">
110  * 
111  * public void testToBean2(){
112  *     String json = "{"ip":"210.75.225.254","country":"\u4e2d\u56fd","area":"\u534e\u5317",
113        "region":"\u5317\u4eac\u5e02","city":"\u5317\u4eac\u5e02","county":"","isp":"\u7535\u4fe1",
114       "country_id":"86","area_id":"100000","region_id":"110000","city_id":"110000",
115      "county_id":"-1","isp_id":"100017"}";
116      
117       Map{@code <String, String>} map = newHashMap();
118         map.put("area_id", "areaId");
119         map.put("region_id", "regionId");
120         map.put("city_id", "cityId");
121         map.put("county_id", "countyId");
122         map.put("country_id", "countryId");
123         map.put("isp_id", "ispId");
124         
125         JsonToJavaConfig jsonToJavaConfig=new JsonToJavaConfig(IpInfoEntity.class, new CustomJavaIdentifierTransformer(map));
126 
127  *     IpInfoEntity ipInfoEntity = JsonUtil.toBean(data, jsonToJavaConfig);
128  *     //.....
129  * }
130  * 
131  * </pre>
132  * 
133  * </blockquote>
134  *
135  * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
136  * @see JavaIdentifierTransformer
137  * @see UncapitalizeJavaIdentifierTransformer
138  * @see <a href="https://github.com/venusdrogon/feilong-json/issues/1">issue 1</a>
139  * @since 1.11.0
140  */
141 public class CustomJavaIdentifierTransformer extends JavaIdentifierTransformer{
142 
143     /** The property name convert map. */
144     private final Map<String, String> propertyNameConvertMap;
145 
146     //---------------------------------------------------------------
147 
148     /**
149      * Instantiates a new map java identifier transformer.
150      *
151      * @param map
152      *            the map
153      */
154     public CustomJavaIdentifierTransformer(Map<String, String> map){
155         Validate.notEmpty(map, "map can't be null/empty!");
156         this.propertyNameConvertMap = map;
157     }
158 
159     //---------------------------------------------------------------
160     /*
161      * (non-Javadoc)
162      * 
163      * @see net.sf.json.util.JavaIdentifierTransformer#transformToJavaIdentifier(java.lang.String)
164      */
165     @Override
166     public String transformToJavaIdentifier(String propertyName){
167         return defaultIfNullOrEmpty(propertyNameConvertMap.get(propertyName), propertyName);
168     }
169 
170 }
