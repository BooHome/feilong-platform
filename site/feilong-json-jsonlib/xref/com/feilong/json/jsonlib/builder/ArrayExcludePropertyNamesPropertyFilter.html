20  /**
21   * 排除属性过滤器.
22   * 
23   * json 转java 的时候,想排除某个属性
24   * 
25   * 比如 taobao 接口返回的数据, 有老的废弃属性 county (可能早期由于单词写错了. 为了兼容老代码)
26   * 
27   * <pre>
28   * {@code
29   * {
30   *             "country": "中国",
31   *             "country_id": "CN",
32   *             "area": "华北",
33   *             "area_id": "100000",
34   *             "region": "北京市",
35   *             "region_id": "110000",
36   *             "city": "北京市",
37   *             "city_id": "110100",
38   *             "county": "",
39   *             "county_id": "-1",
40   *             "isp": "科技网",
41   *             "isp_id": "1000114",
42   *             "ip": "210.75.225.254"
43   *         }
44   * }
45   * </pre>
46   * 
47   * 
48   * 但是我们的java bean 没有这个属性
49   * 
50   * 导致转换的时候, 由于 使用了 PropertyStrategyWrapper
51   * 
52   * <pre>
53   * {@code
54   *   public void setProperty(Object bean,String key,Object value){
55   *         try{
56   *             propertySetStrategy.setProperty(bean, key, value);
57   *         }catch (Exception e){
58   *             LOGGER.warn(e.getMessage(), e);
59   *         }
60   *     }
61   * }
62   * </pre>
63   * 
64   * 会提示 warn 级别的日志, 很烦人
65   *

100 class ArrayExcludePropertyNamesPropertyFilter implements PropertyFilter{
101 
102     /** The property names. */
103     private final String[] propertyNames;
104 
105     //---------------------------------------------------------------
106 
107     /**
108      * The Constructor.
109      *
110      * @param propertyNames
111      *            the property names
112      */
113     public ArrayExcludePropertyNamesPropertyFilter(String...propertyNames){
114         this.propertyNames = propertyNames;
115     }
116 
117     //---------------------------------------------------------------
118 
119     /*
120      * (non-Javadoc)
121      * 
122      * @see net.sf.json.util.PropertyFilter#apply(java.lang.Object, java.lang.String, java.lang.Object)
123      */
124     @Override
125     public boolean apply(Object source,String name,Object value){
126         // [source] the owner of the property
127         // [name] the name of the property
128         // [value] the value of the property
129         return org.apache.commons.lang3.ArrayUtils.contains(propertyNames, name);
130     }
131 }
