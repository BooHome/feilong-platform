63      /**
64       * 转换value的值.
65       *
66       * @param <T>
67       *            the generic type
68       * @param value
69       *            the value
70       * @param jsonToJavaConfig
71       *            the json to java config
72       * @return 如果 value 是 {@link JSONNull#getInstance()} ,那么返回null,<br>
73       *         如果null == jsonToJavaConfig 或者 null == jsonToJavaConfig.getRootClass() 返回value,<br>
74       *         否则,使用 {@link JsonUtil#toBean(Object, JsonToJavaConfig)} 转成对应的bean
75       */
76      @SuppressWarnings("unchecked")
77      static <T> T transformerValue(Object value,JsonToJavaConfig jsonToJavaConfig){
78          if (JSONNull.getInstance().equals(value)){
79              return null;
80          }
81          //如果rootClass是null,表示不需要转换
82          boolean noRootClass = null == jsonToJavaConfig || null == jsonToJavaConfig.getRootClass();
83          return noRootClass ? (T) value : JsonUtil.<T> toBean(value, jsonToJavaConfig);
84      }

87      /**
88       * 把实体Bean、Map对象、数组、列表集合转换成{@link JSON}.
89       * 
90       * <h3>说明:</h3>
91       * 
92       * <blockquote>
93       * <ol>
94       * <li>如果 <code>null==jsonConfig</code>,将使用 {@link #DEFAULT_JAVA_TO_JSON_CONFIG}</li>
95       * 
96       * <li>
97       * 
98       * <p>
99       * 以下类型将转成{@link JSONArray}:
100      * </p>
101      * 
102      * <ul>
103      * <li>如果是字符串,当是以"["符号开头,"]"符号结尾的时候</li>
104      * <li>数组 obj.getClass().isArray()||obj instanceof Object[]</li>
105      * <li>集合 obj instanceof Collection</li>
106      * <li>枚举 obj instanceof Enum</li>
107      * <li>迭代器 obj instanceof Iterator</li>
108      * </ul>
109      * 
110      * </li>
111      * <li>其他类型转成 {@link JSONObject}</li>
112      * </ol>
113      * </blockquote>
114      *
115      * @param obj
116      *            可以是数组,字符串,枚举,集合,map,Java bean,Iterator等类型,内部自动识别转成{@link JSONArray}还是{@link JSONObject}
117      * @param jsonConfig
118      *            the json config
119      * @return the json
120      * @see net.sf.json.JSONArray#fromObject(Object, JsonConfig)
121      * @see net.sf.json.JSONObject#fromObject(Object, JsonConfig)
122      * @see net.sf.json.util.JSONUtils#isArray(Object)
123      * @see java.lang.Class#isEnum()
124      * @see net.sf.json.JsonConfig#registerJsonValueProcessor(Class, JsonValueProcessor)
125      * @see org.apache.commons.collections4.IteratorUtils#toList(Iterator)
126      * @see org.apache.commons.collections4.IteratorUtils#toList(Iterator, int)
127      * @see net.sf.json.JSONSerializer#toJSON(Object)
128      */
129     static JSON toJSON(Object obj,JsonConfig jsonConfig){
130         JsonConfig useJsonConfig = defaultIfNull(jsonConfig, DEFAULT_JAVA_TO_JSON_CONFIG);
131 
132         if (isNeedConvertToJSONArray(obj)){
133             Object arrayJsonObject = obj instanceof Iterator ? IteratorUtils.toList((Iterator<?>) obj) : obj;
134             return toJSONArray(arrayJsonObject, useJsonConfig);
135         }
136         return toJSONObject(obj, useJsonConfig);
137     }

141     /**
142      * 是否需要转成 {@link JSONArray}类型.
143      * 
144      * <h3>以下类型将转成{@link JSONArray}:</h3>
145      * <blockquote>
146      * <ol>
147      * <li>如果是字符串,当是以"["符号开头,"]"符号结尾的时候</li>
148      * <li>数组 obj.getClass().isArray()||obj instanceof Object[]</li>
149      * <li>集合 obj instanceof Collection</li>
150      * <li>枚举 obj instanceof Enum</li>
151      * <li>迭代器 obj instanceof Iterator</li>
152      * </ol>
153      * </blockquote>
154      *
155      * @param obj
156      *            the obj
157      * @return true, if is need convert to JSON array
158      * @see net.sf.json.JSONArray#_fromJSONTokener(net.sf.json.util.JSONTokener, JsonConfig)
159      * @see net.sf.json.util.JSONUtils#isArray(Object)
160      */
161     private static boolean isNeedConvertToJSONArray(Object obj){
162         if (obj instanceof String){
163             String str = (String) obj;
164             if (str.startsWith("[") && str.endsWith("]")){// [] 格式的字符串 
165                 return true;
166             }
167         }
168         return JSONUtils.isArray(obj) || //obj.getClass().isArray() || obj instanceof Collection || obj instanceof Object[]
169                         obj instanceof Enum || // obj.getClass().isEnum()这么写 null会报错// object' is an Enum. Use JSONArray instead
170                         obj instanceof Iterator;
171     }

177     /**
178      * 将 <code>obj</code>转成 {@link JSONArray}.
179      *
180      * @param obj
181      *            Accepts JSON formatted strings, arrays, Collections and Enums.
182      * @param useJsonConfig
183      *            如果是null,将使用 {@link #DEFAULT_JSON_CONFIG_INSTANCE}
184      * @return the JSON array
185      * @see net.sf.json.JSONArray#fromObject(Object, JsonConfig)
186      */
187     static JSONArray toJSONArray(Object obj,JsonConfig useJsonConfig){
188         return JSONArray.fromObject(obj, defaultIfNull(useJsonConfig, DEFAULT_JSON_CONFIG_INSTANCE));
189     }

211     /**
212      * 是否允许被json format的类型.
213      * 
214      * <h3>如果是以下类型是被允许的:</h3>
215      * <blockquote>
216      * <ol>
217      * <li>{@link ClassUtils#isPrimitiveOrWrapper(Class)}</li>
218      * <li>{@link String}</li>
219      * <li>{@link ObjectUtil#isArray(Object)}</li>
220      * <li>{@link ClassUtil#isInstanceAnyClass(Object, Class...)}</li>
221      * </ol>
222      * </blockquote>
223      *
224      * @param <V>
225      *            the value type
226      * @param value
227      *            the value
228      * @param allowClassTypes
229      *            the allow class types
230      * @return true, if checks if is allow format type
231      */
232     static <V> boolean isAllowFormatType(V value,Class<?>...allowClassTypes){
233         if (null == value){//null 是可以 format的
234             return true;
235         }
236 
237         //---------------------------------------------------------------
238         Class<?> klassClass = value.getClass();
239         return ClassUtils.isPrimitiveOrWrapper(klassClass) //
240                         || String.class == klassClass //
241                         || ObjectUtil.isArray(value)//XXX 数组一般 是可以转换的 
242                         || ClassUtil.isInstanceAnyClass(value, allowClassTypes)//
243         ;
244     }
